73 uint16_t LM35::readTemperature()
74 {
75  uint16_t temperatureLast = 0;
76  uint16_t temperature = 0;
77 
78  // Take multiple temperature samples with a maximum of LM35_MAX_SAMPLES
79  for (uint8_t i = 0; i < LM35_MAX_SAMPLES; i++) {
80  // Analog range: 0 .. 1.1V = 0 .. 1023 ADC value
81  // For example, LM35 analog signal of 0.182 V equals to 18.2 degree celsius
82  // 1024 / 1.1V = 931 ratio between analog sample and output temperature
83  // This calculation is multiplied by 1000 first to avoid expensive floating
84  // point instructions.
85  temperature = ((uint32_t)analogRead(_pin) * 1000) / 931;
86 
87  if (temperatureLast == temperature) {
88  // Two identical temperatures read, return
89  return temperature;
90  }
91 
92  // Different temperatures, take another sample
93  temperatureLast = temperature;
94  }
95 
96  // Return the last measured temperature
97  return temperatureLast;
98 }
